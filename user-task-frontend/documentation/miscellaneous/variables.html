updateUser
Default value : async ( id: string, userData: Partial<UserPayload> ): Promise<UserPayload> => { const response = await fetch(`${API_BASE_URL}/users/${id}`, { method: "PUT", // or "PATCH" if partial update headers: { "Content-Type": "application/json" }, body: JSON.stringify(userData), }); if (!response.ok) { throw new Error("Network response was not ok"); } return response.json(); }
